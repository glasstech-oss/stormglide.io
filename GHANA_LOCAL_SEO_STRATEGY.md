### **6.2: Schema Markup for Local Services**

```json
{
  "@context": "https://schema.org",
  "@type": "LocalBusiness",
  "name": "StormGlide Technologies",
  "description": "Custom software development company in Ghana",
  "url": "https://stormglide.vercel.app",
  "telephone": "+233XXX-XXX-XXX",
  "areaServed": "GH",
  "address": {
    "@type": "PostalAddress",
    "streetAddress": "[Your Accra Address]",
    "addressLocality": "Accra",
    "addressRegion": "Greater Accra",
    "postalCode": "[ZIP]",
    "addressCountry": "GH"
  },
  "geo": {
    "@type": "GeoCoordinates",
    "latitude": "5.6037",
    "longitude": "-0.1870"
  },
  "sameAs": [
    "https://facebook.com/stormglide",
    "https://linkedin.com/company/stormglide",
    "https://twitter.com/stormglide"
  ],
  "priceRange": "GHS 2000-500000",

    "@type": "OpeningHoursSpecification",

    "opens": "08:00",
    "closes": "17:00"
  }
}
```
